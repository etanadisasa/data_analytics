d = open('Deuteronomy5.txt', 'r')
import operator, time, string
start = time.time()
deutero = {}
for line in d:
    line=line.split()
    for word in line:
        word = word.lower()
        new_word = word.translate(string.maketrans ("",""), string.punctuation)
        if new_word in deutero:
            deutero[new_word]+=1
        else:
            deutero[new_word]=1
